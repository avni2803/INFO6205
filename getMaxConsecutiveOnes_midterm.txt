class Solution {
    public int getMaxConsecutiveOnes(int[] nums) {
        
    int count = 0;
//maximum count
    int mCount = 0; 
    for(int i = 0; i < nums.length; i++) {
      if(nums[i] == 1) {
        // Increment the count ones  by 1.
        count += 1;

      } else {
        // Find the maximum 

        mCount = Math.max(mCount, count);
        // Resetting the count of 1

        count = 0;
      }
    }
    return Math.max(mCount, count);
  }
}
